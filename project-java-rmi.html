Project 02

Java RMI

Java RMI Distributed Systems Remote Interface Client-Server

Overview

Implementation of a distributed vowel-counting service using Java RMI (Remote Method Invocation). The project demonstrates how objects can invoke methods on remote JVMs transparently, using a registry-based discovery mechanism. The server exposes a compterVoyelles() method which clients call as if it were local — a core concept of distributed computing.

Key Features

  • Remote interface extending java.rmi.Remote with declared RemoteException
  • Server registers service under logical name "SV" via Naming.rebind()
  • Client performs registry lookup and receives a transparent stub
  • Vowel counting logic handles upper/lowercase and null inputs safely
  • Interactive client loop allows multiple remote calls in one session

Source Code

import java.rmi.Remote;
import java.rmi.RemoteException;

/**
 * Interface distante pour le service de comptage de voyelles
 * Doit hériter de Remote — toutes les méthodes déclarent RemoteException
 */
public interface ServiceVoyelle extends Remote {

    /**
     * Compte le nombre de voyelles dans une chaîne (a, e, i, o, u, y)
     * @param texte  La chaîne à analyser
     * @return       Nombre de voyelles trouvées
     * @throws RemoteException En cas d'erreur de communication
     */
    public int compterVoyelles(String texte) throws RemoteException;
}
import java.rmi.RemoteException;
import java.rmi.server.UnicastRemoteObject;

/**
 * Implémentation du service — étend UnicastRemoteObject
 * pour être exporté comme objet distant
 */
public class ServeurVoyelle extends UnicastRemoteObject
                             implements ServiceVoyelle {

    public ServeurVoyelle() throws RemoteException {
        super();
    }

    @Override
    public int compterVoyelles(String texte) throws RemoteException {
        if (texte == null || texte.isEmpty()) return 0;

        int compteur = 0;
        String min = texte.toLowerCase();

        for (int i = 0; i < min.length(); i++) {
            char c = min.charAt(i);
            if (c == 'a' || c == 'e' || c == 'i' ||
                c == 'o' || c == 'u' || c == 'y') {
                compteur++;
            }
        }

        System.out.println("Requête : \"" + texte + "\" → " + compteur + " voyelles");
        return compteur;
    }
}
import java.rmi.Naming;
import java.rmi.registry.LocateRegistry;

/**
 * Démarre le registre RMI, instancie le service et le publie
 */
public class ServeurMain {

    public static void main(String[] args) {
        try {
            // 1 — Créer le registre sur le port par défaut 1099
            LocateRegistry.createRegistry(1099);
            System.out.println("✓ Registre RMI créé sur le port 1099");

            // 2 — Instancier l'objet serveur
            ServeurVoyelle objetServeur = new ServeurVoyelle();

            // 3 — Publier sous le nom logique "SV"
            Naming.rebind("rmi://localhost:1099/SV", objetServeur);
            System.out.println("✓ Service publié → rmi://localhost:1099/SV");
            System.out.println("Serveur en attente de clients...");

        } catch (Exception e) {
            e.printStackTrace();
        }
    }
}
import java.rmi.registry.LocateRegistry;
import java.rmi.registry.Registry;
import java.util.Scanner;

/**
 * Client RMI — se connecte au registre, récupère le stub
 * et appelle compterVoyelles() à distance
 */
public class ClientVoyelle {

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        try {
            // 1 — Récupérer le registre RMI
            Registry registre = LocateRegistry.getRegistry("localhost", 1099);

            // 2 — Récupérer le stub via lookup
            ServiceVoyelle stub = (ServiceVoyelle) registre.lookup("SV");
            System.out.println("✓ Service trouvé — stub récupéré\n");

            // 3 — Boucle d'appels distants
            boolean continuer = true;
            while (continuer) {
                System.out.print("Entrez une chaîne : ");
                String chaine = scanner.nextLine();

                int resultat = stub.compterVoyelles(chaine); // appel distant

                System.out.println("→ Voyelles : " + resultat + "\n");

                System.out.print("Continuer ? (o/n) : ");
                String rep = scanner.nextLine().trim().toLowerCase();
                continuer = rep.equals("o") || rep.equals("oui");
            }

        } catch (Exception e) {
            e.printStackTrace();
        } finally {
            scanner.close();
        }
    }
}
